An asynchronous client library for the KeeeX local API

kxapi = require('kxapi');

kxapi.getToken("APP Name", function(err, resp){
    if(!err){
        console.log("Successfully authorized to access keeex :)");
        // Now you can do whatever you want to
    }
});
